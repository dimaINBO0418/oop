#include "stdafx.h"
#include <fstream>
#include <iostream>
#include <string>
using namespace std;

int main()
{
	setlocale(LC_ALL, "rus");
	string s;
	ifstream file("fff.txt");

	if (file.is_open()){ // вызов метода is_open()
		cout << "Все ОК! Файл открыт!\n\n" << endl;
	for (file >> s; !file.eof(); file >> s) 
		cout << s << endl;
	}
	else
	{
		cout << "Файл не открыт!\n\n" << endl;
		return -1;
	}
	system("pause");
	return 0;
}
